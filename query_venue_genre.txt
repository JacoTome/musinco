?event ?sl ?genre ?venue

where {
  [a musico:MusicianParticipation] musico:involved_event ?event;
   musico:played_musical_work ?o.
  ?o mo:genre ?genre.
  ?sl a musico:SelfLearning .
  bind (str(?event) as ?event_string).
  
  bind (str(?sl) as ?sl_string) .
  bind (strafter(?event_string,http://www.semanticweb.org/jaco/ontologies/2023/7/musinco/MusicalEvent/) as ?ev_Res).	
  bind (strafter(?sl_string,http://www.semanticweb.org/jaco/ontologies/2023/7/musinco/SelfLearningSession/) as ?sl_res).
  ?sl schema:address ?venue.
    
  filter (?ev_Res = ?sl_res)
